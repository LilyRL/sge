float box(vec2 position, vec2 halfSize, float cornerRadius) {
   position = abs(position) - halfSize + cornerRadius;
   return length(max(position, 0.0)) + min(max(position.x, position.y), 0.0) - cornerRadius;
}

vec4 drawBox(vec2 center, vec2 size, float cornerRadius, vec4 color, vec2 fragCoord) {
    vec2 position = (2.0 * (fragCoord - center) - iResolution.xy) / iResolution.y;
    vec2 halfSize = size / 2.0;
    float d = box(position, halfSize, cornerRadius);

    return vec4(1.0) - vec4(sign(d)) * color;
}

void mainImage(out vec4 fragColor, in vec2 fragCoord) {
    vec2 center = vec2(100.0, 0.0);
    vec2 size = vec2(1.0, 0.5);
    float cornerRadius = 0.2; // clamp to between 0 and (shortest side length / 2)
	fragColor = drawBox(center, size, cornerRadius, vec4(1.0, 1.0, 1.0, 1.0), fragCoord);
}
